def _derive_gate_facts(task: dict, dispatch: dict | None) -> list[str]:
    facts: list[str] = []
    if task.get("run_ref") and task.get("assigned_to"):
        facts.append("dispatched")
    if dispatch and dispatch.get("dispatch_state") == "reclaimed":
        facts.append("reclaimed")
    if task.get("review_ref"):
        facts.append("review_ready")
    if task.get("status") == "done":
        facts.append("accepted")
    return facts


def _promotion_blockers(task: dict, gate_facts: list[str]) -> list[str]:
    blockers: list[str] = []
    if "dispatched" in gate_facts and not task.get("review_ref"):
        blockers.append("missing_review_ref")
    if task.get("evidence_required") and "review_ready" not in gate_facts:
        blockers.append("evidence_not_ready")
    return blockers
```

def write_worker_utilization_summary(root: Path) -> str:
    runs = _load_dispatch_runs(root)
    by_worker: dict[str, dict[str, int]] = {}
    for run in runs:
        stats = by_worker.setdefault(run["worker_id"], {"dispatches": 0, "reclaims": 0, "completed": 0})
        stats["dispatches"] += 1
        if run.get("dispatch_state") == "reclaimed":
            stats["reclaims"] += 1
        if run.get("dispatch_state") == "completed":
            stats["completed"] += 1
    return _write_markdown_summary(root, by_worker)
```

def write_handoff_index(root: Path, task_id: str) -> str:
    task = _load_active_or_done_task(root, task_id)
    refs = {
        "run_ref": task.get("run_ref"),
        "review_ref": task.get("review_ref"),
        "handoff_refs": task.get("handoff_refs", []),
    }
    output = root / ".omo" / "evidence" / "handoffs" / f"{task_id}.md"
    output.write_text(_render_handoff_markdown(task_id, refs), encoding="utf-8")
    return str(output.relative_to(root))
